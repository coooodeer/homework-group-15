uint32 sbox(uint32 b) {
	uint32 a[4];

	a[0] = (b >> 24) & 0xff;
	a[1] = (b >> 16) & 0xff;
	a[2] = (b >> 8) & 0xff;
	a[3] = b & 0xff;

	b = (Sbox[a[0]] << 24) | (Sbox[a[1]] << 16) | (Sbox[a[2]] << 8) | Sbox[a[3]];

	return b;
}
uint32 L(uint32 b)
{
	return b ^ loopleft(b, 2) ^ loopleft(b, 10) ^ loopleft(b, 18) ^ loopleft(b, 24);

uint32 Lkey(uint32 b)
{
	return b ^ loopleft(b, 13) ^ loopleft(b, 23);
}
uint32 loopleft(uint32 a, short length) {
	return (a << length) + (a >> (32 - length));
}
uint32 T(uint32 b)
{
	b = sbox(b);
	b = L(b);
	return b;
}

	return b;

		k[i] = k[i] ^ FK[i];

void genRoundKey(uint32 k[])

	for (int i = 4; i < 36; i += 1)
	{
		k[i] = k[i - 4] ^ Tkey(k[i - 3] ^ k[i - 2] ^ k[i - 1] ^ CK[i - 4]);
		//printf("第 %u 个轮密钥是%x\n", i - 4, k[i]);
	}

}

{
	return a ^ T(b ^ c ^ d ^ roundk);
}
void encrypt(uint32 k[4], uint32 m[4], uint32 c[4])
{
	//printf("------------加密中------------\n");
	uint32 M[36] = { 0 };
	uint32 K[36] = { 0 };

	for (int i = 0; i < 4; i += 1)
	{
		M[i] = m[i];
		K[i] = k[i];
	}
	initKey(K);
	genRoundKey(K);

		//printf("%x ", c[i]);
	}
}

		uint8 b1 = (uint8)((temp >> 24) & 0xff);
		uint8 b2 = (uint8)((temp >> 16) & 0xff);
		uint8 b3 = (uint8)((temp >> 8) & 0xff);
		uint8 b4 = (uint8)(temp & 0xff);
		state[i] = ((uint32)SubByte(b1, mode)<<24)| ((uint32)SubByte(b2, mode) << 16)| ((uint32)SubByte(b3, mode) << 8)| (uint32)SubByte(b4, mode);
	}	
}
void ShiftRows(uint32 state[4], int mode) {
	if (mode == Encrypt) {
		uint32 temp1 = state[0];
		uint32 temp2 = state[1];
		uint32 temp3 = state[2];
		uint32 temp4 = state[3];
		state[0] = (temp1 & 0xff000000) | (temp2 & 0x00ff0000) | (temp3 & 0x0000ff00) | (temp4 & 0x000000ff);
		state[1] = (temp2 & 0xff000000) | (temp3 & 0x00ff0000) | (temp4 & 0x0000ff00) | (temp1 & 0x000000ff);
		state[2] = (temp3 & 0xff000000) | (temp4 & 0x00ff0000) | (temp1 & 0x0000ff00) | (temp2 & 0x000000ff);
		state[3] = (temp4 & 0xff000000) | (temp1 & 0x00ff0000) | (temp2 & 0x0000ff00) | (temp3 & 0x000000ff);
		}
	else{
		uint32 temp1 = state[0];
		uint32 temp2 = state[1];
		uint32 temp3 = state[2];
		uint32 temp4 = state[3];
		state[0] = (temp1 & 0xff000000) | (temp4 & 0x00ff0000) | (temp3 & 0x0000ff00) | (temp2 & 0x000000ff);
		state[1] = (temp2 & 0xff000000) | (temp1 & 0x00ff0000) | (temp4 & 0x0000ff00) | (temp3 & 0x000000ff);
		state[2] = (temp3 & 0xff000000) | (temp2 & 0x00ff0000) | (temp1 & 0x0000ff00) | (temp4 & 0x000000ff);
		state[3] = (temp4 & 0xff000000) | (temp3 & 0x00ff0000) | (temp2 & 0x0000ff00) | (temp1 & 0x000000ff);
	}
}
void MixColumns(uint32 state[4], int mode) {
	if (mode == Encrypt) {
		for (int i = 0; i < 4; i++) {
			uint32 temp = state[i];
			uint8 c0 = (temp >> 24) & 0xff;
			uint8 c1 = (temp >> 16) & 0xff;
			uint8 c2 = (temp >> 8) & 0xff;
			uint8 c3 = temp & 0xff;
			uint8 r0 = TE[1][c0] ^ TE[2][c1] ^ TE[0][c2] ^ TE[0][c3];
			uint8 r1 = TE[0][c0] ^ TE[1][c1] ^ TE[2][c2] ^ TE[0][c3];
			uint8 r2 = TE[0][c0] ^ TE[0][c1] ^ TE[1][c2] ^ TE[2][c3];
			uint8 r3 = TE[2][c0] ^ TE[0][c1] ^ TE[0][c2] ^ TE[1][c3];
			state[i] = ((uint32)r0 << 24) | ((uint32)r1 << 16) | ((uint32)r2 << 8) | (uint32)r3;
		}
	}
	else {
		for (int i = 0; i < 4; i++) {
			uint32 temp = state[i];
			uint8 c0 = (temp >> 24) & 0xff;
			uint8 c1 = (temp >> 16) & 0xff;
			uint8 c2 = (temp >> 8) & 0xff;
			uint8 c3 = temp & 0xff;
			uint8 r0 = TD[3][c0] ^ TD[1][c1] ^ TD[2][c2] ^ TD[0][c3];
			uint8 r1 = TD[0][c0] ^ TD[3][c1] ^ TD[1][c2] ^ TD[2][c3];
			uint8 r2 = TD[2][c0] ^ TD[0][c1] ^ TD[3][c2] ^ TD[1][c3];
			uint8 r3 = TD[1][c0] ^ TD[2][c1] ^ TD[0][c2] ^ TD[3][c3];
			state[i] = ((uint32)r0 << 24) | ((uint32)r1 << 16) | ((uint32)r2 << 8) | (uint32)r3;
		}
	}
}

void AESencrypt(uint32 state[4], uint32 key[4]) {
	uint32 W[44] = { 0 };
	generateRoundKey(key, W);
	AddRoundKey(state, 0, W);
	for (int r = 1; r < 10; r += 1) {
		SubBytes(state, Encrypt);
		ShiftRows(state, Encrypt);
		MixColumns(state, Encrypt);
		AddRoundKey(state, r, W);
	}
	SubBytes(state, Encrypt);
	ShiftRows(state, Encrypt);	
	AddRoundKey(state, 10, W);
}
void AESdecrypt(uint32 state[4], uint32 key[4]) {
	uint32 W[44] = { 0 };
	generateRoundKey(key, W);
	AddRoundKey(state, 10, W);
	for (int r = 9; r > 0; r -= 1) {
		ShiftRows(state, Decrypt);
		SubBytes(state, Decrypt);
		AddRoundKey(state, r, W);
		MixColumns(state, Decrypt);
	}
	ShiftRows(state, Decrypt);
	SubBytes(state, Decrypt);
	AddRoundKey(state, 0, W);
}
